    2 module Tuples where
    3 {- 
    4 This module is part of chapter 1 of The Ray Tracer Challenge
    5 
    6 This is the library for tuples, one of the backbones of a Raytracer. Used in all kinds of calculations.
    7 
    8 For example when light hits an object. 
    9 
   10 Some keynotions are
   11   vector = tuple (x y z 0) 
   12    w = 0
   13   point = tuple (x y z 1)
   14    w = 1
   15 
   16   vector is used for motions and forcer, hence the w = 0
   17   point is visable and rather small object, therefore the w = 1
   18 -}   
   19 epsilon = 0.00001
   20 
   21 isPoint :: (Fractional a, Num a, Eq a) => Tuple a -> Bool
   22 isPoint tuple = getW tuple == 1.0
   23 
   24 point :: (Fractional a, Num a) => a -> a -> a -> Tuple a 
   25 point x y z = Tuple x y z 1.0
   26 
   27 isVector :: (Num a,Fractional a, Eq a) => Tuple a -> Bool
   28 isVector tuple = getW tuple == 0.0
   29 
   30 vector :: (Fractional a, Num a) => a -> a -> a -> Tuple a
   31 vector x y z = Tuple x y z 0.0
   32 
   33 tuple :: Num a =>  a -> a -> a -> a -> Tuple a
   34 tuple = Tuple
   35 
   36 magnitude :: (Floating a, Num a) => Tuple a -> a
   37 magnitude (Tuple x y z w) = sqrt ((x^2) + (y^2) + (z^2) + (w^2))
   38 
   39 normalize :: (Floating a, Num a) => Tuple a -> Tuple a
   40 normalize t@(Tuple x y z w) = tuple (x / magnitude t) (y / magnitude t) (z / magnitude t) (w / magnitude t) 
   41 
   42 dot :: (Floating a, Num a) => Tuple a -> Tuple a -> a
   43 dot (Tuple x1 y1 z1 w1) (Tuple x2 y2 z2 w2) = (x1 * x2) + (y1 * y2) + (z1 * z2) + (w1 * w2)  
   44 
   45 cross :: (Eq a,Floating a, Num a) => Tuple a -> Tuple a -> Tuple a
   46 cross (Tuple x1 y1 z1 0) (Tuple x2 y2 z2 0) = vector ((y1 * z2) - (z1 * y2)) ((z1 * x2) - (x1 * z2)) ((x1 * y2) - (y1 * x2)) 
   47 
   48 reflect :: (Tuple Double, Tuple Double) -> Tuple Double
   49 reflect (v,n) = v - n * pure (2 * dot v n)
   50 
   51 data Tuple a = Tuple { getX :: a, getY :: a, getZ :: a, getW :: a }
   52  deriving (Show)
   53 
   54 instance Eq (Tuple Double) where
   55  Tuple x1 y1 z1 w1 == Tuple x2 y2 z2 w2 = (compare x1 x2) && (compare y1 y2) && (compare z1 z2) && (compare w1 w2) 
   56   where compare a b = abs (a - b) < epsilon
   57 
   58 instance Functor Tuple where
   59  fmap f (Tuple x y z w) = Tuple (f x) (f y) (f z) (f w) 
   60 
   61 instance Applicative Tuple where
   62  pure x = Tuple x x x x
   63  (Tuple x1 y1 z1 w1) <*> (Tuple x2 y2 z2 w2) = Tuple (x1 x2) (y1 y2) (z1 z2) (w1 w2)
   64 
   65 instance (Num a, Fractional a) => Num (Tuple a) where
   66  a + b = (fmap (+) a) <*> b
   67  a - b = (fmap (-) a) <*> b
   68  negate a = vector 0 0 0 - a
   69  a * b = (fmap (*) a) <*> b
   70 
   71 instance (Fractional a) => Fractional (Tuple a) where
   72  a / b = (fmap (/) a) <*> b
